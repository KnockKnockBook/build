Filters

Where logic is true
# The basic process for filtering a DF is to:
# 1 write a true/false statement 
# 2 use the statement to filter the DF to cases where the DF is true 
# you do this by adding the true/false statement to the DF[True_False_Statement] 

# Create a True/False statement for a DF column
(DF['B'] == 'b')

# adding a True/False statement to a DF column
DF[ (DF['B'] == 'b') ]
Where column IS equal to a value
DF[( DF['B'] == 'b' )]
Where column NOT equal to a value
DF[( DF['B'] != 'b' )]
Where column is NULL
DF[( DF['has_nulls'].isnull() )]
Where column NOT NULL
DF[( DF['has_nulls'].notnull() )]
Where multiple conditions - AND
# use '&' to seporate your true/false statements
DF[ ( DF['B'] == 'b' )  &  ( DF['A'] == 2 ) ]
Where multiple conditions - OR
# use '|' to seporate your true/false statements
DF[ ( DF['B'] == 'b' )  |  ( DF['B'] == 'c' ) ]
Where column IN a list of values
# use .isin(['your','list', 'of', 'values'])
# create a true/false condition using a list of values
DF['B'].isin(['a','c'])

# use it to filter a DF
DF[ ( DF['B'].isin(['a','c']) ) ]
Where column NOT IN list of values
# use the ~ before your true/false condition
DF[ ~( DF['B'].isin(['a','c']) ) ]

Create new columns with logical conditions
# Create a new column where condition is true (new column will be 'NaN' where it's not true)
DF.loc[DF['B'] == 'b', 'new_column_name'] = 'value_if_true'
# Create a second statement where where condition is NOT true to update all the other values 
DF.loc[DF['B'] != 'b', 'new_column_name'] = 'value_if_false'
